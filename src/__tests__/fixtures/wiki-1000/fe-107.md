# Zustand Patterns and Middleware

## Store Definition

```typescript
import { create } from 'zustand';
import { devtools, persist, immer } from 'zustand/middleware';

interface TodoStore {
  todos: Todo[];
  filter: 'all' | 'active' | 'done';
  addTodo: (text: string) => void;
  toggleTodo: (id: string) => void;
  setFilter: (filter: TodoStore['filter']) => void;
}

const useTodoStore = create<TodoStore>()(
  devtools(
    persist(
      immer((set) => ({
        todos: [],
        filter: 'all',
        addTodo: (text) =>
          set((state) => {
            state.todos.push({ id: crypto.randomUUID(), text, done: false });
          }),
        toggleTodo: (id) =>
          set((state) => {
            const todo = state.todos.find((t) => t.id === id);
            if (todo) todo.done = !todo.done;
          }),
        setFilter: (filter) => set({ filter }),
      })),
      { name: 'todo-store' }
    )
  )
);
```

## Selectors

Subscribe to specific slices to prevent unnecessary re-renders:

```typescript
// Only re-renders when todos change, not when filter changes
const todos = useTodoStore((state) => state.todos);

// Derived selector
const activeTodos = useTodoStore((state) =>
  state.todos.filter((t) => !t.done)
);
```

## Testing

Zustand stores are plain functions, easy to test without React:

```typescript
const store = useTodoStore.getState();
store.addTodo('Test item');
expect(useTodoStore.getState().todos).toHaveLength(1);
```
