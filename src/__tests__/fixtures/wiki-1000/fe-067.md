# WebSocket Integration in React

## Connection Management

Establish WebSocket connections at the application level, not per component:

```typescript
class WebSocketManager {
  private ws: WebSocket | null = null;
  private listeners = new Map<string, Set<(data: any) => void>>();
  private reconnectAttempts = 0;

  connect(url: string) {
    this.ws = new WebSocket(url);

    this.ws.onmessage = (event) => {
      const { type, payload } = JSON.parse(event.data);
      this.listeners.get(type)?.forEach((cb) => cb(payload));
    };

    this.ws.onclose = () => this.scheduleReconnect(url);
  }

  subscribe(type: string, callback: (data: any) => void) {
    if (!this.listeners.has(type)) {
      this.listeners.set(type, new Set());
    }
    this.listeners.get(type)!.add(callback);
    return () => this.listeners.get(type)?.delete(callback);
  }

  private scheduleReconnect(url: string) {
    const delay = Math.min(1000 * 2 ** this.reconnectAttempts, 30000);
    setTimeout(() => {
      this.reconnectAttempts++;
      this.connect(url);
    }, delay);
  }
}
```

## React Hook

```typescript
function useWebSocket<T>(type: string): T | null {
  const [data, setData] = useState<T | null>(null);
  const wsManager = useContext(WebSocketContext);

  useEffect(() => {
    return wsManager.subscribe(type, setData);
  }, [wsManager, type]);

  return data;
}
```

## When to Use WebSockets

Real-time features: chat, notifications, collaborative editing, live dashboards. For occasional updates, server-sent events (SSE) or polling are simpler alternatives.
