# API Deprecation Headers

## Standard Headers

```
Deprecation: true
Sunset: Sat, 01 Jun 2026 00:00:00 GMT
Link: </v3/orders>; rel="successor-version"
```

## Implementation

```typescript
function deprecationMiddleware(
  sunsetDate: string,
  successorUrl?: string
): Middleware {
  return (req, res, next) => {
    res.setHeader('Deprecation', 'true');
    res.setHeader('Sunset', new Date(sunsetDate).toUTCString());
    
    if (successorUrl) {
      res.setHeader('Link', `<${successorUrl}>; rel="successor-version"`);
    }
    
    // Track usage for sunset planning
    logger.info({
      deprecated_endpoint: req.path,
      client_id: req.auth?.clientId,
      api_key_prefix: req.auth?.apiKeyPrefix,
    });
    
    next();
  };
}

// Apply to deprecated routes
app.get('/v1/orders', deprecationMiddleware('2026-06-01', '/v2/orders'), v1OrderHandler);
```
