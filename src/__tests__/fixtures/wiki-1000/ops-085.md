# Terraform Conditional Resources

## Count-Based Conditionals

Use `count` to conditionally create resources:

```hcl
variable "enable_monitoring" {
  type    = bool
  default = true
}

resource "aws_cloudwatch_metric_alarm" "cpu" {
  count = var.enable_monitoring ? 1 : 0

  alarm_name  = "high-cpu"
  metric_name = "CPUUtilization"
  namespace   = "AWS/EC2"
  statistic   = "Average"
  threshold   = 80
  period      = 300
}
```

## For_Each Conditionals

```hcl
variable "environments" {
  type = map(object({
    instance_type = string
    create_rds    = bool
  }))
  default = {
    dev     = { instance_type = "t3.small", create_rds = false }
    staging = { instance_type = "t3.medium", create_rds = true }
    prod    = { instance_type = "m5.large", create_rds = true }
  }
}

resource "aws_db_instance" "main" {
  for_each = { for k, v in var.environments : k => v if v.create_rds }

  identifier     = "db-${each.key}"
  instance_class = "db.${each.value.instance_type}"
  engine         = "postgres"
}
```

## Dynamic Blocks

For optional nested blocks within a resource:

```hcl
resource "aws_security_group" "app" {
  name = "app-sg"

  dynamic "ingress" {
    for_each = var.allowed_cidrs
    content {
      from_port   = 443
      to_port     = 443
      protocol    = "tcp"
      cidr_blocks = [ingress.value]
    }
  }
}
```
